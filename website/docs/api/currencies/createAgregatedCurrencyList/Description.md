# Unit definitions



### CurrencyList

```ts

interface CurrencyList {
  contains: (currency: Currency) => boolean;
  getCurrencies(): CurrencyMap;
  subUnitFor(currency: Currency): number;
}


```


### CurrencyMap


```ts

 type CurrencyMap<C> = {
  [code: string]: C;
};


```


### Currency

```ts

type Currency = string | AnyCurrency;

```

interface AnyCurrency extends CurrencyUnit {

}
